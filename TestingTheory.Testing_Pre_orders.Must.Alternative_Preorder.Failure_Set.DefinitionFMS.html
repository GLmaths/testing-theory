Definition Failure `{
  gLtsP : @gLts P A H, CC : Countable PreAct, @FinitaryAbsAction P T FinA PreAct A H Φ 𝝳 _ gLtsEqT _ _ }
  (p : P) (s : trace A) (G : gset PreAct) :=
  p ⇓ s -> exists p', p ⟹[s] p' /\ forall β μ, ( 𝝳 ∘ Φ ) β ∈ G -> dual μ β -> blocking β
  -> ¬ exists p0, p' ⟹{μ} p0.
